-- DASHBOARD 1: Case Status Distribution
SELECT 
    case_status,
    COUNT(*) AS case_count,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 1) AS percentage
FROM MISSING_PERSONS
GROUP BY case_status
ORDER BY case_count DESC;

-- DASHBOARD 2: Trafficking Hotspots
SELECT 
    last_seen_province,
    COUNT(*) AS total_cases,
    SUM(CASE WHEN suspected_trafficking='Y' THEN 1 ELSE 0 END) AS trafficking_cases,
    ROUND(AVG(age), 1) AS avg_age,
    ROUND(SUM(CASE WHEN suspected_trafficking='Y' THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 1) AS trafficking_rate
FROM MISSING_PERSONS
WHERE case_status = 'ACTIVE'
GROUP BY last_seen_province
ORDER BY trafficking_rate DESC;

-- DASHBOARD 3: Alert Performance
SELECT 
    alert_type,
    COUNT(*) AS total_alerts,
    ROUND(AVG(match_score), 1) AS avg_score,
    SUM(CASE WHEN alert_status='CONFIRMED' THEN 1 ELSE 0 END) AS confirmed,
    ROUND(SUM(CASE WHEN alert_status='CONFIRMED' THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 1) AS accuracy_rate
FROM ALERTS
GROUP BY alert_type
ORDER BY accuracy_rate DESC;

-- DASHBOARD 4: Agency Workload
SELECT 
    a.agency_name,
    a.agency_type,
    COUNT(mp.report_id) AS total_cases,
    SUM(CASE WHEN mp.case_status='ACTIVE' THEN 1 ELSE 0 END) AS active_cases,
    SUM(CASE WHEN mp.case_status='FOUND' THEN 1 ELSE 0 END) AS resolved_cases,
    ROUND(SUM(CASE WHEN mp.case_status='FOUND' THEN 1 ELSE 0 END) * 100.0 / 
          NULLIF(COUNT(mp.report_id), 0), 1) AS resolution_rate
FROM AGENCIES a
LEFT JOIN MISSING_PERSONS mp ON a.agency_id = mp.agency_id
GROUP BY a.agency_name, a.agency_type
ORDER BY total_cases DESC;

-- DASHBOARD 5: Executive KPIs
SELECT 
    missing_persons_pkg.get_total_active_cases() AS active_cases,
    missing_persons_pkg.get_trafficking_cases_count() AS trafficking_cases,
    missing_persons_pkg.get_resolution_rate(30) AS resolution_rate_30d,
    missing_persons_pkg.get_resolution_rate(90) AS resolution_rate_90d,
    (SELECT COUNT(*) FROM ALERTS WHERE alert_status='NEW') AS pending_alerts,
    (SELECT COUNT(*) FROM SIGHTINGS WHERE verification_status='PENDING') AS pending_sightings,
    (SELECT ROUND(AVG(credibility_score), 1) FROM SIGHTINGS) AS avg_credibility
FROM DUAL;

-- DASHBOARD 6: Temporal Patterns
SELECT 
    TO_CHAR(last_seen_date, 'DAY') AS day_of_week,
    TO_CHAR(last_seen_date, 'D') AS day_num,
    COUNT(*) AS total_cases,
    SUM(CASE WHEN suspected_trafficking='Y' THEN 1 ELSE 0 END) AS trafficking_cases,
    ROUND(AVG(age), 1) AS avg_age
FROM MISSING_PERSONS
WHERE last_seen_date >= ADD_MONTHS(SYSDATE, -12)
GROUP BY TO_CHAR(last_seen_date, 'DAY'), TO_CHAR(last_seen_date, 'D')
ORDER BY TO_NUMBER(TO_CHAR(last_seen_date, 'D'));

-- DASHBOARD 7: Audit Compliance
SELECT 
    attempt_type,
    attempt_day,
    is_holiday,
    COUNT(*) AS attempt_count,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER(), 1) AS percentage
FROM AUDIT_LOGS
WHERE attempt_type IS NOT NULL
  AND operation_timestamp >= SYSTIMESTAMP - INTERVAL '30' DAY
GROUP BY attempt_type, attempt_day, is_holiday
ORDER BY attempt_count DESC;

    last_seen_province,
    last_seen_district,

-- Dashboard 3: Alert Performance
CREATE OR REPLACE VIEW v_alert_performance AS
SELECT 
    alert_type,
    alert_status,
    priority,
    COUNT(*) AS alert_count,
    ROUND(AVG(match_score), 1) AS avg_score,
    MIN(created_at) AS first_alert,
    MAX(created_at) AS last_alert

GROUP BY alert_type, alert_status, priority;
